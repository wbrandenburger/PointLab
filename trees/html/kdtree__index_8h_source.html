168  void computeBoundingBox(BoundingBox& bbox_)
169  {
170  bbox_.resize(veclen);
171  for (size_t i = 0; i<veclen; ++i) {
172  bbox_[i].low = (ElementType)dataset[0][i];
173  bbox_[i].high = (ElementType)dataset[0][i];
174  }
175  for (size_t k = 1; k<size; ++k) {
176  for (size_t i = 0; i<veclen; ++i) {
177  if (dataset[k][i]<bbox_[i].low) bbox_[i].low = (ElementType)dataset[k][i];
178  if (dataset[k][i]>bbox_[i].high) bbox_[i].high = (ElementType)dataset[k][i];
179  }
180  }
181  }

193  NodePtr divideTree(int left_, int right_, BoundingBox& bbox_)
194  {
195  NodePtr node = new (pool) Node(); // allocate memory
196 
197  /* If too few exemplars remain, then make this a leaf node. */
198  if ((right_ - left_) <= neighbor) {
199  node->child1 = node->child2 = NULL; /* Mark as leaf node. */
200  node->left = left_;
201  node->right = right_;
202 
203  // compute bounding-box of leaf points
204  for (size_t i = 0; i<veclen; ++i) {
205  bbox_[i].low = (ElementType)dataset[vind[left_]][i];
206  bbox_[i].high = (ElementType)dataset[vind[left_]][i];
207  }
208  for (int k = left_ + 1; k<right_; ++k) {
209  for (size_t i = 0; i<veclen; ++i) {
210  if (bbox_[i].low>dataset[vind[k]][i]) bbox_[i].low = (ElementType)dataset[vind[k]][i];
211  if (bbox_[i].high<dataset[vind[k]][i]) bbox_[i].high = (ElementType)dataset[vind[k]][i];
212  }
213  }
214  }
215  else {
216  int idx;
217  int cutfeat;
218  ElementType cutval;
219  middleSplit(&vind[0] + left_, right_ - left_, idx, cutfeat, cutval, bbox_);
220 
221  node->divfeat = cutfeat;
222 
223  BoundingBox left_bbox(bbox_);
224  left_bbox[cutfeat].high = cutval;
225  node->child1 = divideTree(left_, left_ + idx, left_bbox);
226 
227  BoundingBox right_bbox(bbox_);
228  right_bbox[cutfeat].low = cutval;
229  node->child2 = divideTree(left_ + idx, right_, right_bbox);
230 
231  node->divlow = left_bbox[cutfeat].high;
232  node->divhigh = right_bbox[cutfeat].low;
233 
234  for (size_t i = 0; i<veclen; ++i) {
235  bbox_[i].low = std::min(left_bbox[i].low, right_bbox[i].low);
236  bbox_[i].high = std::max(left_bbox[i].high, right_bbox[i].high);
237  }
238  }
239 
240  return node;
241  }

274  void middleSplit(int* ind_, int count_, int& index_, int& cutfeat_, ElementType& cutval_, const BoundingBox& bbox_)
275  {
276  // find the largest span from the approximate bounding box
277  ElementType max_span = bbox_[0].high - bbox_[0].low;
278  cutfeat_ = 0;
279  cutval_ = (bbox_[0].high + bbox_[0].low) / 2;
280  for (size_t i = 1; i<veclen; ++i) {
281  ElementType span = bbox_[i].high - bbox_[i].low;
282  if (span>max_span) {
283  max_span = span;
284  cutfeat_ = i;
285  cutval_ = (bbox_[i].high + bbox_[i].low) / 2;
286  }
287  }
288 
289  // compute exact span on the found dimension
290  ElementType min_elem, max_elem;
291  computeMinMax(ind_, count_, cutfeat_, min_elem, max_elem);
292  cutval_ = (min_elem + max_elem) / 2;
293  max_span = max_elem - min_elem;
294 
295  // check if a dimension of a largest span exists
296  size_t k = cutfeat_;
297  for (size_t i = 0; i<veclen; ++i) {
298  if (i == k) continue;
299  ElementType span = bbox_[i].high - bbox_[i].low;
300  if (span>max_span) {
301  computeMinMax(ind_, count_, i, min_elem, max_elem);
302  span = max_elem - min_elem;
303  if (span>max_span) {
304  max_span = span;
305  cutfeat_ = i;
306  cutval_ = (min_elem + max_elem) / 2;
307  }
308  }
309  }
310  int lim1, lim2;
311  planeSplit(ind_, count_, cutfeat_, cutval_, lim1, lim2);
312 
313  if (lim1>count_ / 2) index_ = lim1;
314  else if (lim2<count_ / 2) index_ = lim2;
315  else index_ = count_ / 2;
316 
317  assert(index > 0 && index < count_);
318  }

329  void planeSplit(int* ind_, int count_, int cutfeat_, ElementType cutval_, int& lim1_, int& lim2_)
330  {
331  int left = 0;
332  int right = count_ - 1;
333  for (;; ) {
334  while (left <= right && dataset[ind_[left]][cutfeat_]<cutval_) ++left;
335  while (left <= right && dataset[ind_[right]][cutfeat_] >= cutval_) --right;
336  if (left>right) break;
337  std::swap(ind_[left], ind_[right]); ++left; --right;
338  }
339 
340  lim1_ = left;
341  right = count_ - 1;
342  for (;; ) {
343  while (left <= right && dataset[ind_[left]][cutfeat_] <= cutval_) ++left;
344  while (left <= right && dataset[ind_[right]][cutfeat_]>cutval_) --right;
345  if (left>right) break;
346  std::swap(ind_[left], ind_[right]); ++left; --right;
347  }
348  lim2_ = left;
349  }

380  ElementType computeInitialDistances(const ElementType* vec_, std::vector<ElementType>& dists_) const
381  {
382  ElementType distsq = 0.0;
383 
384  for (size_t i = 0; i < veclen; ++i) {
385  if (vec_[i] < root_bbox[i].low) {
386  dists_[i] = (vec_[i] - root_bbox[i].low)*(vec_[i] - root_bbox[i].low);
387  distsq += dists_[i];
388  }
389  if (vec_[i] > root_bbox[i].high) {
390  dists_[i] = (vec_[i] - root_bbox[i].high)*(vec_[i] - root_bbox[i].high);
391  distsq += dists_[i];
392  }

409  void searchLevel(ResultSet<ElementType>& result_set_, const ElementType* vec_, const NodePtr node_, ElementType mindistsq_,
410  std::vector<ElementType>& dists_, const float epsError_) const
411  {
412  /* If this is a leaf node, then do check and return. */
413  if ((node_->child1 == NULL) && (node_->child2 == NULL)) {
414  ElementType worst_dist = result_set_.worstDist();
415  for (int i = node_->left; i<node_->right; ++i) {
416  ElementType* point = ordered ? dataset_ordered[i] : dataset[vind[i]];
417  ElementType dist = 0;
418  for (int j = 0; j < veclen; j++) {
419  dist += (vec_[j] - point[j])*(vec_[j] - point[j]);
420  }
421 
422  if (dist<worst_dist) {
423  result_set_.addPoint(dist, vind[i]);
424  }
425  }
426  return;
427  }
428 
429  /* Which child branch should be taken first? */
430  int idx = node_->divfeat;
431  ElementType val = vec_[idx];
432  ElementType diff1 = val - node_->divlow;
433  ElementType diff2 = val - node_->divhigh;
434 
435  NodePtr bestChild;
436  NodePtr otherChild;
437  ElementType cut_dist;
438  if ((diff1 + diff2)<0) {
439  bestChild = node_->child1;
440  otherChild = node_->child2;
441  cut_dist = (val-node_->divhigh)*(val-node_->divhigh);
442  }
443  else {
444  bestChild = node_->child2;
445  otherChild = node_->child1;
446  cut_dist = (val - node_->divlow)*(val - node_->divlow);
447  }
448 
449  /* Call recursively to search next level down. */
450  searchLevel(result_set_, vec_, bestChild, mindistsq_, dists_, epsError_);
451 
452  ElementType dst = dists_[idx];
453  mindistsq_ = mindistsq_ + cut_dist - dst;
454  dists_[idx] = cut_dist;
455  if (mindistsq_*epsError_ <= result_set_.worstDist()) {
456  searchLevel(result_set_, vec_, otherChild, mindistsq_, dists_, epsError_);
457  }
458  dists_[idx] = dst;
459  }
