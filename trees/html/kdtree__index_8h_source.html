299  NodePtr divideTree(NodePtr parent_, int left_, int right_, BoundingBox& bbox_)
300  {
301  NodePtr node = new (pool) Node; // allocate memory
302  node->parent = parent_;
303  /* If too few exemplars remain, then make this a leaf node. */
304  if ((right_ - left_) <= neighbor) {
305  node->child1 = node->child2 = nullptr; /* Mark as leaf node. */
306 
307  node->points = right_ - left_;
308  node->indices= new size_t[node->points];
309  for (size_t i = left_; i < right_; i++) {
310  node->indices[i - left_] = i;
311 
312  dataset_nodes[vind[i]] = node;
313  }
314 
315  // compute bounding-box of leaf points
316  for (size_t i = 0; i<veclen; ++i) {
317  bbox_[i].low = (ElementType)dataset_points[vind[left_]][i];
318  bbox_[i].high = (ElementType)dataset_points[vind[left_]][i];
319  }
320  for (int k = left_ + 1; k<right_; ++k) {
321  for (size_t i = 0; i<veclen; ++i) {
322  if (bbox_[i].low>dataset_points[vind[k]][i]) bbox_[i].low = (ElementType)dataset_points[vind[k]][i];
323  if (bbox_[i].high<dataset_points[vind[k]][i]) bbox_[i].high = (ElementType)dataset_points[vind[k]][i];
324  }
325  }
326  }
327  else {
328  int idx;
329  int cutfeat;
330  ElementType cutval;
331  middleSplit(&vind[0] + left_, right_ - left_, idx, cutfeat, cutval, bbox_);
332 
333  node->divfeat = cutfeat;
334 
335  BoundingBox left_bbox(bbox_);
336  left_bbox[cutfeat].high = cutval;
337  node->child1 = divideTree(node, left_, left_ + idx, left_bbox);
338 
339  BoundingBox right_bbox(bbox_);
340  right_bbox[cutfeat].low = cutval;
341  node->child2 = divideTree(node, left_ + idx, right_, right_bbox);
342 
343  node->divlow = left_bbox[cutfeat].high;
344  node->divhigh = right_bbox[cutfeat].low;
345 
346  for (size_t i = 0; i<veclen; ++i) {
347  bbox_[i].low = std::min(left_bbox[i].low, right_bbox[i].low);
348  bbox_[i].high = std::max(left_bbox[i].high, right_bbox[i].high);
349  }
350  }
351 
352  return node;
353  }

386  void middleSplit(size_t* ind_, int count_, int& index_, int& cutfeat_, ElementType& cutval_, const BoundingBox& bbox_)
387  {
388  // find the largest span from the approximate bounding box
389  ElementType max_span = bbox_[0].high - bbox_[0].low;
390  cutfeat_ = 0;
391  cutval_ = (bbox_[0].high + bbox_[0].low) / 2;
392  for (size_t i = 1; i<veclen; ++i) {
393  ElementType span = bbox_[i].high - bbox_[i].low;
394  if (span>max_span) {
395  max_span = span;
396  cutfeat_ = i;
397  cutval_ = (bbox_[i].high + bbox_[i].low) / 2;
398  }
399  }
400 
401  // compute exact span on the found dimension
402  ElementType min_elem, max_elem;
403  computeMinMax(ind_, count_, cutfeat_, min_elem, max_elem);
404  cutval_ = (min_elem + max_elem) / 2;
405  max_span = max_elem - min_elem;
406 
407  // check if a dimension of a largest span exists
408  size_t k = cutfeat_;
409  for (size_t i = 0; i<veclen; ++i) {
410  if (i == k) continue;
411  ElementType span = bbox_[i].high - bbox_[i].low;
412  if (span>max_span) {
413  computeMinMax(ind_, count_, i, min_elem, max_elem);
414  span = max_elem - min_elem;
415  if (span>max_span) {
416  max_span = span;
417  cutfeat_ = i;
418  cutval_ = (min_elem + max_elem) / 2;
419  }
420  }
421  }
422  int lim1, lim2;
423  planeSplit(ind_, count_, cutfeat_, cutval_, lim1, lim2);
424 
425  if (lim1>count_ / 2) index_ = lim1;
426  else if (lim2<count_ / 2) index_ = lim2;
427  else index_ = count_ / 2;
428 
429  if (index_ < 0 && index_ > count_) {
430  std::cout << "Exit in " << __FILE__ << " in line " << __LINE__ << std::endl;
431  std::exit(EXIT_FAILURE);
432  }
433  }

444  void planeSplit(size_t* ind_, int count_, int cutfeat_, ElementType cutval_, int& lim1_, int& lim2_)
445  {
446  int left = 0;
447  int right = count_ - 1;
448  for (;; ) {
449  while (left <= right && dataset_points[ind_[left]][cutfeat_]<cutval_) ++left;
450  while (left <= right && dataset_points[ind_[right]][cutfeat_] >= cutval_) --right;
451  if (left>right) break;
452  std::swap(ind_[left], ind_[right]); ++left; --right;
453  }
454 
455  lim1_ = left;
456  right = count_ - 1;
457  for (;; ) {
458  while (left <= right && dataset_points[ind_[left]][cutfeat_] <= cutval_) ++left;
459  while (left <= right && dataset_points[ind_[right]][cutfeat_]>cutval_) --right;
460  if (left>right) break;
461  std::swap(ind_[left], ind_[right]); ++left; --right;
462  }
463  lim2_ = left;
464  }

565  void searchLevel(ResultSet<ElementType>& result_set_, const ElementType* vec_, const NodePtr node_, ElementType mindistsq_,
566  std::vector<ElementType>& dists_, const float epsError_) const
567  {
568  /* If this is a leaf node, then do check and return. */
569  if ((node_->child1 == nullptr) && (node_->child2 == nullptr)) {
570  ElementType worst_dist = result_set_.worstDist();
571  for (int i = 0; i<node_->points; ++i) {
572  ElementType* point = ordered ? dataset_points[node_->indices[i]] : dataset_points[vind[node_->indices[i]]];
573 
574  ElementType dist = distance(const_cast<ElementType*>(vec_), point, veclen);
575  if (dist<worst_dist) {
576  result_set_.addPoint(dist, vind[node_->indices[i]]);
577  }
578  }
579  return;
580  }
581 
582  /* Which child branch should be taken first? */
583  int idx = node_->divfeat;
584  ElementType val = vec_[idx];
585  ElementType diff1 = val - node_->divlow;
586  ElementType diff2 = val - node_->divhigh;
587 
588  NodePtr best_child;
589  NodePtr other_child;
590  ElementType cut_dist;
591  if ((diff1 + diff2)<0) {
592  best_child = node_->child1;
593  other_child = node_->child2;
594  cut_dist = distance(val, node_->divhigh);
595  }
596  else {
597  best_child = node_->child2;
598  other_child = node_->child1;
599  cut_dist = distance(val, node_->divlow);
600  }
601 
602  /* Call recursively to search next level down. */
603  if (best_child) {
604  searchLevel(result_set_, vec_, best_child, mindistsq_, dists_, epsError_);
605  }
606 
607  if (other_child) {
608  ElementType dst = dists_[idx];
609  mindistsq_ = mindistsq_ + cut_dist - dst;
610  dists_[idx] = cut_dist;
611  if (mindistsq_*epsError_ <= result_set_.worstDist()) {
612  searchLevel(result_set_, vec_, other_child, mindistsq_, dists_, epsError_);
613  }
614  dists_[idx] = dst;
615  }
616  }
