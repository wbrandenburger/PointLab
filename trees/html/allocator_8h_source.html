103  void* allocateMemory(int size_)
104  {
105  /* Round size up to a multiple of wordsize. The following expression
106  only works for WORDSIZE that is a power of 2, by masking last bits of
107  incremented size to zero.
108  */
109  size_ = (size_ + (wordsize - 1)) & ~(wordsize - 1);
110 
111  /* Check whether a new block must be allocated. Note that the first word
112  of a block is reserved for a pointer to the previous block.
113  */
114  if (size_ > remaining) {
115 
116  wastedMemory += remaining;
117 
118  /* Allocate new storage. */
119  blocksize = (size_ + sizeof(void*) + (wordsize-1) > blocksize) ?
120  size_ + sizeof(void*) + (wordsize-1) : blocksize;
121 
122  // use the standard C malloc to allocate memory
123  void* m = ::malloc(blocksize);
124  if (!m) {
125  fprintf(stderr,"Failed to allocate memory.\n");
126  return NULL;
127  }
128 
129  /* Fill first word of new block with pointer to previous block. */
130  ((void**) m)[0] = base;
131  base = m;
132 
133  int shift = 0;
134  //int shift = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
135 
136  remaining = blocksize - sizeof(void*) - shift;
137  loc = ((char*)m + sizeof(void*) + shift);
138  }
139  void* rloc = loc;
140  loc = (char*)loc + size_;
141  remaining -= size_;
142 
143  usedMemory += size_;
144 
145  return rloc;
146  }
