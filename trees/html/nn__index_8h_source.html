106  void knnSearch(const Matrix<ElementType>& queries_,
107  Matrix<size_t>& indices_,
108  Matrix<ElementType>& dists_,
109  size_t knn_,
110  const TreeParams& params_)
111  {
112  assert(queries_.cols == veclen());
113  assert(indices_.rows >= queries_.rows);
114  assert(dists_.rows >= queries_.rows);
115  assert(indices_.cols >= knn_);
116  assert(dists_.cols >= knn_);
117 
118  Threadpool pool(params_.cores);
119 
120  for (int i = 0; i < queries_.rows; i++) {
121  while (!pool.runTask(boost::bind(&NNIndex<ElementType>::knnSearchThreadpool,
122  this,
123  boost::ref(queries_),
124  boost::ref(indices_),
125  boost::ref(dists_),
126  knn_,
127  boost::ref(params_),
128  i)));
129  }
130 
131  pool.shutdown();

144  void knnSearchThreadpool(const Matrix<ElementType>& queries_,
