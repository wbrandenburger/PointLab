92  template<typename T> struct BigAnyPolicy : TypedBaseAnyPolicy<T>
93  {
94  virtual void staticDelete(void** x)
95  {
96  if (* x) delete (* reinterpret_cast<T**>(x)); *x = NULL;
97  }
98  virtual void copyFromValue(void const* src, void** dest)
99  {
100  *dest = new T(*reinterpret_cast<T const*>(src));
101  }
102  virtual void clone(void* const* src, void** dest)
103  {
104  *dest = new T(**reinterpret_cast<T* const*>(src));
105  }
106  virtual void move(void* const* src, void** dest)
107  {
108  (*reinterpret_cast<T**>(dest))->~T();
109  **reinterpret_cast<T**>(dest) = **reinterpret_cast<T* const*>(src);
110  }
111  virtual void* getValue(void** src) { return *src; }
112  virtual const void* getValue(void* const * src) { return *src; }
113  virtual void print(std::ostream& out, void* const* src) { out << *reinterpret_cast<T const*>(*src); }
114  };
